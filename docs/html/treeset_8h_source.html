182template <typename T, typename Compare> inline
183void TreeSetIter<T, Compare>::inorder_traverse_to_leftmost_node(
184TreeSet<T, Compare>::sp_node from_node) {
185 while (from_node != nullptr) {
186 _next_node_stack.push(from_node);
187 from_node = from_node->left;
188 }
189
190 if (!_next_node_stack.empty()) {
191 _current_node = _next_node_stack.top();
192 _next_node_stack.pop();
193 } else {
194 _current_node = nullptr;
195 }

417template <typename T, typename Compare> inline bool
418TreeSet<T, Compare>::sanity_check(const sp_node &n) const {
419 // Only perform sanity check if T has std::numeric_limits.
420 // Use this TreeSet's Compare fn to determine minval & maxval to use for check
421 if (std::numeric_limits<T>::is_specialized) {
422 // Initial guess at min/max values
423 T minval = std::numeric_limits<T>::min();
424 T maxval = std::numeric_limits<T>::max();
425
426 // If guess was wrong, swap values
427 if (!_cmp(minval, maxval)) {
428 std::swap(minval, maxval);
429 }
430
431 return sanity_check(n, minval, maxval);
432 }
433
434 // Intentionality not performing any sanity checks for other T and Compare types
435 return true;

456template <typename T, typename Compare> inline
457bool TreeSet<T, Compare>::add(T value) {
458 assert(sanity_check(_root));
459
460 if (size() == 0) {
461 _root = std::make_shared<node>(value);
462 _size = 1;
463
464 assert(sanity_check(_root));
465
466 return true;
467 }
468
469 sp_node n = _root;
470
471 while (n != nullptr) {
472 if (value == n->value) { // value already exists
473 return false;
474 } else if (_cmp(value, n->value)) { // attempt add to left subtree
475 if (n->left == nullptr) {
476 n->left = std::make_shared<node>(value);
477 _size++;
478 return true;
479 } else {
480 n = n->left;
481 }
482 } else { // attempt add to right subtree
483 if (n->right == nullptr) {
484 n->right = std::make_shared<node>(value);
485 _size++;
486 return true;
487 } else {
488 n = n->right;
489 }
490 }
491 }
492
493 assert(sanity_check(_root));
494
495 return false;
496}

518template <typename T, typename Compare> inline
519bool TreeSet<T, Compare>::del(T value) {
520 assert(sanity_check(_root));
521
522 if (size() == 0)
523 return false;
524
525 sp_node n = _root;
526 sp_node parent = nullptr;
527
528 while (n != nullptr) {
529 if (value == n->value) { // found value to delete
530 if (parent == nullptr) { // we need to update the _root
531 // so merge the _root's children into the new _root
532 _root = merge(n->left, n->right);
533 } else { // no need to update the _root
534 // we are deleting n, so first merge n's children
535 sp_node merged_children = merge(n->left, n->right);
536
537 // now replace n with its merged children
538 parent->replace_child(n, merged_children);
539 }
540
541 _size--;
542 return true;
543 } else if (_cmp(value, n->value)) { // attempt delete from left subtree
544 parent = n;
545 n = n->left;
546 } else { // attempt delete from right subtree
547 parent = n;
548 n = n->right;
549 }
550 }
551
552 assert(sanity_check(_root));
553
554 return false;
555}
